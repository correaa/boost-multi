\begin{frame}
\frametitle{As a proposal}

Why not proposed for Standard?
\begin{itemize}
  \item Can't promise ABI compatibility
  \item Controversial idioms (some hacks)
  \item Too restrictive for compatibility with exotic 3rd parties (Thrust, BLAS, CUDA, cuLibs)
  \item Enough overlap with \texttt{std::span} and \texttt{std::mdarray}
  \item Long time horizon
\end{itemize}
 
Why proposed to Boost?
\begin{itemize}
  \item Ideal balance between risk (innovation) and requirements (quality, guidelines)
  \item \emph{Peer reviewed}
  \item Still good distribution model and visibility
  \item No need for compiler-magic (although new language features would help)
  \item Moderate time horizon
  \item Can eventually benefit from low level Boost tools (Boost.Hana, Boost.Assert, Boost.Align, Boost.STLInterfaces) 
  \item Peer interaction with Boost.Serialization, Boost.JSON, Boost.QVM, Boost.YAP
\end{itemize}
\end{frame}

\begin{frame}{Fundamental View/Layout operations}
\begin{itemize}
\item Rotate (indices)\\
  \lstinline|A[i][j][k]...[x][y][z]| $\to$ \lstinline|A[z][i][j]...[x][y]|
\item Transpose (first two indices) \\
  \lstinline|A[i][j][k]...| $\to$ \lstinline|A[j][i][k]...|
\end{itemize}
\end{frame}

\begin{frame}{Recursive Indexing}
  \lstinline|A[i]| removes one dimension \\
  \lstinline|A[i][j]| removes two dimensions \\
  \ldots
\end{frame}

\begin{frame}{Positional indexing}
  \lstinline|A[3, _]| $\equiv$ \lstinline|A[3]| \\
  \lstinline|A[_, 3]| $\equiv$ \lstinline|A.rotated()[3].unrotated()| \\
\end{frame}

\begin{frame}{Implement positional indices}
\begin{itemize}
\item Recursive Indexing (leading dimensions)
\item Positional Indexing (leading dimensions)
  \lstinline|A[_, i]| = \lstinline|A.rotated()[i].unrotated()| \\
\end{itemize}
\end{frame}


\begin{frame}[fragile]{Algorithms}
  Take a 64x64x64 array ($\sim 250$k elements) \lstinline|A|, apply a compute-bound op.

  \only<1>{
    \begin{lstlisting}
for(auto&& plane : A) {
  for(auto&& row : plane) {
    for(auto&& elem : row) {
      elem = op(elem);
    }
  }
}
    \end{lstlisting}
  }

  \only<2>{
    \begin{lstlisting}
std::for_each(A.begin(), A.end(), [](auto&& plane) {
  for(auto&& row : plane) {
    for(auto&& elem : row) {
      elem = op(elem);
    }
  }
});
    \end{lstlisting}
  }

  timing: \textbf{0.0026 sec}
\end{frame}
